东软学习小组成员：夜枫
题目：
算术表达式有前缀表示法、中缀表示法和后缀表示法等形式。前缀表达式指二元运算符位于两个运算数之前，例如2+3*(7-4)+8/4的前缀表达式是：+ + 2 * 3 - 7 4 / 8 4。请设计程序计算前缀表达式的结果值。

输入格式:
输入在一行内给出不超过30个字符的前缀表达式，只包含+、-、*、/以及运算数，不同对象（运算数、运算符号）之间以空格分隔。

输出格式:
输出前缀表达式的运算结果，保留小数点后1位，或错误信息ERROR。

输入样例:

```cpp
+ + 2 * 3 - 7 4 / 8 4
```

输出样例:

```cpp
13.0
```

c++代码
```cpp
#include<iostream>
#include<stack>
#include <string>
#include<iomanip>
#include<sstream>
using namespace std;
stack <double> s;
bool AllisNum(string str) {//检测str是否为数字
	stringstream sin(str);
	double d;
	char c;
	if (!(sin >> d))
	{
		return false;
	}
	if (sin >> c)
	{
		return false;
	}
	return true;
}
double cal(double a, double b, char op) {
	if (op == '-') return a - b;
	else if (op == '+') return a + b;
	else if (op == '*') return a * b;
	else if (b == 0) {
		cout << "ERROR";
		exit(0);
	}
	return a / b;
}
int main(void) {
	string str;
	string ss[200];
	int nu = 0;
	double a, t1, t2, res;
	getline(cin, str);//读取一行
	istringstream iss(str);//以空格作为分隔，放进ss容器
	while (iss >> str) {
		ss[nu] = str;
		nu++;
	}
	for (int i = nu- 1; i >= 0; --i) {
		if (AllisNum(ss[i])) {//检测是否为数字
			a = stod(ss[i]); string 转为 double  “2.3”-->2.3
			s.push(a);//放进栈
		}
		else {//取出两个 运算并返回栈
			t1 = s.top(); s.pop();
			t2 = s.top(); s.pop();
			s.push(cal(t1, t2, ss[i][0]));
		}
	}
	if (!s.empty()) {
		res = s.top();
		printf("%.1lf\n", res);
	}
	return 0;
}
```
